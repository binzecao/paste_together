安装 Jenkins：
docker pull jenkins/jenkins:2.245-alpine

运行镜像：
mkdir -p /shared_files/shared_projects/jenkins_home
chmod 777 /shared_files/shared_projects/jenkins_home

docker run --name jenkins \
-d \
-v /shared_files/shared_projects/jenkins_home:/var/jenkins_home \
-p 27733:8080 \
-p 50000:50000 \
--cpus=1 \
-m 256M --memory-swap=1024M \
jenkins/jenkins:2.245-alpine
